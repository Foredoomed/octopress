public static void quickSort1(int[] array, int p, int q) {</p>

<p>if (p &gt; q) {
    return;
  }</p>

<p>// 选取左端点为pivot
  int pivot = array[p];
  int j = p;</p>

<p>for (int i = j + 1; i &lt;= q; i++) {
    if (array[i] &lt; pivot) {
      // 如果小于pivot则与j+1位置上数交换
      swap(array, ++j, i);</p>

<pre><code>} }
</code></pre>

<p>// 交换pivot与j位置上的数
  swap(array, p, j);</p>

<p>// 迭代左右子数组
  quickSort1(array, p, j - 1);
  quickSort1(array, j + 1, q);
}</p>

<p>public static void swap(int[] array, int i, int j) {
  int temp = array[i];
  array[i] = array[j];
  array[j] = temp;
}
