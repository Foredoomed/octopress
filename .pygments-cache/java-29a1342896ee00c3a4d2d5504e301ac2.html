public static void quickSort2(int[] array, int p, int q) {

  int i = p, j = q;
		
  // 选中间数为pivot
  int pivot = array[i + (q - p) / 2];

  while (i <= j) {
			
    while (array[i] < pivot) {
      i++;
    }
			
    while (array[j] > pivot) {
      j--;
    }

    // 交换小于pivot的数和大于pivot的数
    if (i <= j) {
      swap(array, i, j);
      i++;
      j--;
	 }
  }
		
  // 迭代
  if (p < j)
    quickSort2(array, p, j);
  if (i < q)
    quickSort2(array, i, q);
}
